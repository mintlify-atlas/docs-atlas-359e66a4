## Quick Example

Let's start with a simple example. The text `"This"` (4 bytes) generates the following CID:

```
Content: This
Length:  4 bytes
CID:     AAAAAAAEVGhpcw
```

Breaking down the CID:
- `AAAAAAAE` - Length prefix (8 characters encoding the number 4)
- `VGhpcw` - Base64url encoding of "This"

<Note>
For content ≤ 64 bytes, the CID contains the actual content (base64url-encoded). For larger content, it contains a SHA-512 hash instead.
</Note>

## Computing a CID

Choose your preferred language to get started:

<CodeGroup>
```python Python
import base64
import hashlib

def to_base64url(data: bytes) -> str:
    """Encode bytes as base64url (RFC 4648 Section 5)"""
    return base64.urlsafe_b64encode(data).decode().rstrip("=")

def encode_length(length: int) -> str:
    """Encode content length as 8-character base64url string"""
    return to_base64url(length.to_bytes(6, "big"))

def compute_cid(content: bytes) -> str:
    """Compute the content identifier for given content"""
    prefix = encode_length(len(content))
    
    if len(content) <= 64:
        # Small content: store inline
        suffix = to_base64url(content)
    else:
        # Large content: store SHA-512 hash
        suffix = to_base64url(hashlib.sha512(content).digest())
    
    return prefix + suffix

# Example usage
content = b"This"
cid = compute_cid(content)
print(f"CID: {cid}")
# Output: CID: AAAAAAAEVGhpcw
```

```javascript JavaScript
const crypto = require('crypto');

function toBase64Url(buffer) {
  return buffer
    .toString('base64')
    .replace(/\+/g, '-')
    .replace(/\//g, '_')
    .replace(/=+$/g, '');
}

function encodeLength(length) {
  const bytes = Buffer.alloc(6);
  bytes.writeUIntBE(length, 0, 6);
  return toBase64Url(bytes);
}

function computeCid(content) {
  const prefix = encodeLength(content.length);
  
  const suffix = content.length <= 64
    ? toBase64Url(content)
    : toBase64Url(crypto.createHash('sha512').update(content).digest());
  
  return prefix + suffix;
}

// Example usage
const content = Buffer.from('This');
const cid = computeCid(content);
console.log(`CID: ${cid}`);
// Output: CID: AAAAAAAEVGhpcw
```

```go Go
package main

import (
	"crypto/sha512"
	"encoding/base64"
	"encoding/binary"
	"fmt"
)

func toBase64URL(data []byte) string {
	return base64.RawURLEncoding.EncodeToString(data)
}

func encodeLength(length int) string {
	var buffer [8]byte
	binary.BigEndian.PutUint64(buffer[:], uint64(length))
	return toBase64URL(buffer[2:])
}

func computeCID(content []byte) string {
	prefix := encodeLength(len(content))

	var suffix string
	if len(content) <= 64 {
		suffix = toBase64URL(content)
	} else {
		hash := sha512.Sum512(content)
		suffix = toBase64URL(hash[:])
	}

	return prefix + suffix
}

func main() {
	content := []byte("This")
	cid := computeCID(content)
	fmt.Printf("CID: %s\n", cid)
	// Output: CID: AAAAAAAEVGhpcw
}
```

```bash Bash
#!/bin/bash

# Compute CID using OpenSSL and base64
function compute_cid() {
    local content="$1"
    local length=${#content}
    
    # Encode length as base64url
    local length_hex=$(printf '%012x' $length)
    local length_b64=$(echo -n $length_hex | xxd -r -p | base64 | tr '+/' '-_' | tr -d '=')
    
    if [ $length -le 64 ]; then
        # Small content: encode inline
        local content_b64=$(echo -n "$content" | base64 | tr '+/' '-_' | tr -d '=')
        echo "${length_b64}${content_b64}"
    else
        # Large content: hash with SHA-512
        local hash_b64=$(echo -n "$content" | openssl dgst -sha512 -binary | base64 | tr '+/' '-_' | tr -d '=')
        echo "${length_b64}${hash_b64}"
    fi
}

# Example usage
cid=$(compute_cid "This")
echo "CID: $cid"
# Output: CID: AAAAAAAEVGhpcw
```

## Understanding the Examples

### Small Content Example (≤ 64 bytes)

For the content `"This"` (4 bytes):

<Steps>
  <Step title="Compute the length prefix">
    - Content length: 4 bytes
    - Convert to 6-byte big-endian: `0x000000000004`
    - Base64url encode: `AAAAAAAE`
  </Step>
  
  <Step title="Encode the content">
    - Content: `This`
    - Base64url encode: `VGhpcw`
  </Step>
  
  <Step title="Concatenate">
    - CID: `AAAAAAAE` + `VGhpcw` = `AAAAAAAEVGhpcw`
  </Step>
</Steps>

### Large Content Example (> 64 bytes)

For content larger than 64 bytes, we use SHA-512:

<Steps>
  <Step title="Compute the length prefix">
    - Content length: 1024 bytes
    - Convert to 6-byte big-endian: `0x000000000400`
    - Base64url encode: `AAAAEAA`
  </Step>
  
  <Step title="Hash the content">
    - Compute SHA-512 hash of content (64 bytes)
    - Base64url encode the hash (86 characters)
  </Step>
  
  <Step title="Concatenate">
    - CID: `AAAAEAA` + (86-char hash) = 94-character CID
  </Step>
</Steps>

## Verifying a CID

To verify that content matches a given CID, simply recompute the CID and compare:

<CodeGroup>
```python Python
def verify_cid(content: bytes, expected_cid: str) -> bool:
    """Verify content matches the expected CID"""
    computed_cid = compute_cid(content)
    return computed_cid == expected_cid

# Example verification
content = b"This"
expected = "AAAAAAAEVGhpcw"

if verify_cid(content, expected):
    print("✓ Content verified!")
else:
    print("✗ Content does not match CID")
```

```javascript JavaScript
function verifyCid(content, expectedCid) {
  const computedCid = computeCid(content);
  return computedCid === expectedCid;
}

// Example verification
const content = Buffer.from('This');
const expected = 'AAAAAAAEVGhpcw';

if (verifyCid(content, expected)) {
  console.log('✓ Content verified!');
} else {
  console.log('✗ Content does not match CID');
}
```

```go Go
func verifyCID(content []byte, expectedCID string) bool {
	computedCID := computeCID(content)
	return computedCID == expectedCID
}

// Example verification
func main() {
	content := []byte("This")
	expected := "AAAAAAAEVGhpcw"

	if verifyCID(content, expected) {
		fmt.Println("✓ Content verified!")
	} else {
		fmt.Println("✗ Content does not match CID")
	}
}
```
</CodeGroup>

<Warning>
**Always verify content before using it.** It's easier for an attacker to lie about content than to engineer a hash collision. Verification protects against both malicious and accidental mismatches.
</Warning>

## Common Pitfalls

<Warning>
**Base64url vs Base64**: Make sure you use base64url encoding (RFC 4648 Section 5), which replaces `+` with `-` and `/` with `_`, and omits padding `=` characters.
</Warning>

<Warning>
**Length Encoding**: The length must be encoded as a 6-byte (48-bit) big-endian integer before base64url encoding, not as an ASCII string.
</Warning>

<Warning>
**Small Content Threshold**: Content of exactly 64 bytes can be stored inline. The threshold is ≤ 64 bytes, not < 64 bytes.
</Warning>
